\begin{exercise}
    求极限 $\displaystyle \lim_{x\to 0} \frac{\sin 3x}{x}$.
\end{exercise}
\begin{solution}
    利用重要极限，$\displaystyle \lim_{x\to 0} \frac{\sin 3x}{x}=3\lim_{x\to 0}\frac{\sin 3x}{3x}=3\times 1=3$.
\end{solution}
